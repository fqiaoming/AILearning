### 一、STAR生成器

```python
# resume_optimizer.py
"""简历优化工具"""

from dataclasses import dataclass
from typing import List, Optional

@dataclass
class ProjectSTAR:
    """项目STAR描述"""
    situation: str  # 情境
    task: str  # 任务
    action: List[str]  # 行动
    result: List[str]  # 结果
    tech_stack: List[str]  # 技术栈
    
    def generate(self) -> str:
        """生成项目描述"""
        
        description = f"""
**项目背景：**
{self.situation}

**我的职责：**
{self.task}

**技术方案：**
"""
        
        for tech in self.tech_stack:
            description += f"• {tech}\n"
        
        description += "\n**核心成果：**\n"
        
        for result in self.result:
            description += f"• ✅ {result}\n"
        
        if self.action:
            description += "\n**技术亮点：**\n"
            for action in self.action:
                description += f"• {action}\n"
        
        return description.strip()

class ResumeOptimizer:
    """简历优化器"""
    
    @staticmethod
    def quantify_impact(before: float, after: float, unit: str = "%") -> str:
        """
        量化影响
        
        Args:
            before: 优化前
            after: 优化后
            unit: 单位
        
        Returns:
            描述
        """
        if after > before:
            increase = ((after - before) / before) * 100
            if unit == "%":
                return f"从{before}%提升到{after}%（+{after-before}%）"
            elif unit == "x":
                times = after / before
                return f"从{before}提升到{after}（{times:.1f}x提升）"
            elif unit == "ms":
                decrease = ((before - after) / before) * 100
                return f"从{before}ms降到{after}ms（降低{decrease:.0f}%）"
            else:
                return f"从{before}{unit}提升到{after}{unit}"
        else:
            decrease = ((before - after) / before) * 100
            return f"从{before}{unit}降到{after}{unit}（降低{decrease:.0f}%）"
    
    @staticmethod
    def highlight_keywords(text: str) -> List[str]:
        """
        提取关键词
        
        Args:
            text: 描述文本
        
        Returns:
            关键词列表
        """
        # AI相关关键词
        ai_keywords = [
            "LangChain", "RAG", "Agent", "GPT", "Claude",
            "Milvus", "Chroma", "FAISS", "Pinecone",
            "微调", "LoRA", "量化", "Prompt",
            "向量数据库", "Embedding", "检索"
        ]
        
        # 技术关键词
        tech_keywords = [
            "Python", "FastAPI", "Django", "React",
            "Docker", "Kubernetes", "Redis", "PostgreSQL",
            "异步", "并发", "缓存", "优化"
        ]
        
        # 业务关键词
        business_keywords = [
            "准确率", "QPS", "响应时间", "成本",
            "用户满意度", "转化率", "效率"
        ]
        
        all_keywords = ai_keywords + tech_keywords + business_keywords
        
        found_keywords = []
        for keyword in all_keywords:
            if keyword in text:
                found_keywords.append(keyword)
        
        return found_keywords

# 使用示例
project = ProjectSTAR(
    situation="公司有10万+份内部文档，员工查找困难，每天500+咨询",
    task="作为技术负责人，从0到1设计并实现企业级RAG问答系统",
    action=[
        "实现混合检索算法，召回率提升12%",
        "优化向量检索性能，QPS从500提升到1000",
        "设计完整监控体系（Prometheus + Grafana）"
    ],
    result=[
        "准确率92%（超过目标85%）",
        "P95响应时间1.8s（目标<2s）",
        "支持10万文档，1000+ QPS",
        "自动解决率80%，咨询量下降70%",
        "2个月按时交付上线"
    ],
    tech_stack=[
        "架构：LangChain + Milvus + GPT-4",
        "文档处理：支持PDF/Word/Markdown，智能分块",
        "检索优化：混合检索（向量+BM25）+ 重排序",
        "性能优化：Redis缓存 + 异步处理"
    ]
)

print(project.generate())
print()

# 量化示例
optimizer = ResumeOptimizer()
print("量化示例：")
```

```markdown
**正确做法：**
• 诚实标注熟练程度（精通/熟练/了解）
• 明确说明自己的角色（负责/参与）
• 数据务必真实

---

## ❌ 禁忌2：格式混乱

**错误：**
• 多种字体混用
• 颜色花哨
• 排版不对齐
• 2M+的PDF

**后果：**
• 给人不专业的印象
• HR阅读困难
• 直接淘汰

**正确做法：**
• 统一字体（推荐：微软雅黑/苹方）
• 黑色为主，少用颜色
• 对齐整齐
• PDF<500KB

---

## ❌ 禁忌3：流水账

**错误：**
"负责公司AI项目开发，参与需求讨论，编写代码，
测试，部署，维护，完成领导交办的任务。"

**问题：**
• 没有重点
• 没有成果
• 看不出能力

**正确做法：**
• 用STAR法则
• 突出成果
• 量化数据

---

## ❌ 禁忌4：无关信息过多

**错误：**
• 小学、初中经历
• 过多的个人爱好
• 身高体重
• 政治面貌（非必要）
• 婚姻状况

**正确做法：**
• 只写相关信息
• 技术岗：技术+项目为主
• 1-2页足够

---

## ❌ 禁忌5：技能罗列

**错误：**
"精通：Python, Java, C++, Go, JavaScript, 
      React, Vue, Angular, Django, Flask,
      MySQL, PostgreSQL, MongoDB, Redis..."

**问题：**
• 不可信
• 没重点
• 哪个都不精

**正确做法：**
• 分级标注
• 突出核心技能
• 提供证明

---

## ❌ 禁忌6：负面信息

**错误：**
• "因公司裁员离职"
• "与领导理念不合"
• "公司倒闭"
• "项目失败"

**正确做法：**
• 正面描述
• "寻求更大发展空间"
• "完成阶段性目标"
• 面试时再详细解释

---

## ❌ 禁忌7：错别字

**错误：**
• "项目开发"→"相木开发"
• "负责"→"付责"
• 中英文标点混用

**后果：**
• 体现不细心
• 降低专业性

**正确做法：**
• 仔细检查3遍
• 使用拼写检查
• 让朋友帮忙看

---

## ❌ 禁忌8：缺少联系方式

**错误：**
• 没有手机号
• 邮箱错误
• GitHub链接失效

**正确做法：**
• 提供有效联系方式
• 确保随时能联系到
• 链接可访问

---

## ❌ 禁忌9：期望薪资不合理

**错误：**
• 0-3年经验，期望50K+
• 或者期望过低

**正确做法：**
• 了解市场行情
• 根据自己水平定价
• 可以写范围（如：20-30K）

## ❌ 禁忌10：投递不匹配岗位

**错误：**
• 算法岗简历投应用岗

• 简历与JD完全不匹配

**正确做法：**
• 针对岗位调整简历
• 突出匹配的技能
```
