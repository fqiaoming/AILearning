```python
from dataclasses import dataclass

class CareerPlanner:
    """职业规划器"""
    
    @staticmethod
    def print_skill_tree():
        """打印技能树"""
        print("="*80)
        print("AI工程师技能树")
        print("="*80)
        print()
        
        for level, skills in AI_ENGINEER_SKILLS.items():
            print(f"### {level.value}")
            print()
            
            # 按类别分组
            by_category = {}
            for skill in skills:
                if skill.category not in by_category:
                    by_category[skill.category] = []
                by_category[skill.category].append(skill)
            
            # 打印每个类别
            for category, category_skills in by_category.items():
                print(f"**{category.value}：**")
                for skill in category_skills:
                    stars = "⭐" * skill.importance
                    print(f"  • {skill.name} {stars}")
                print()
            
            print()
    
    @staticmethod
    def generate_learning_plan(current_level: Level, target_level: Level):
        """生成学习计划"""
        print("="*80)
        print(f"学习计划：{current_level.value} → {target_level.value}")
        print("="*80)
        print()
        
        # 获取当前级别和目标级别之间的所有技能
        level_order = [Level.JUNIOR, Level.MIDDLE, Level.SENIOR, Level.EXPERT]
        current_idx = level_order.index(current_level)
        target_idx = level_order.index(target_level)
        
        if target_idx <= current_idx:
            print("目标级别应该高于当前级别")
            return
        
        # 收集需要学习的技能
        skills_to_learn = []
        for i in range(current_idx + 1, target_idx + 1):
            level = level_order[i]
            skills_to_learn.extend(AI_ENGINEER_SKILLS[level])
        
        # 按类别分组
        by_category = {}
        for skill in skills_to_learn:
            if skill.category not in by_category:
                by_category[skill.category] = []
            by_category[skill.category].append(skill)
        
        # 打印学习计划
        for category, skills in by_category.items():
            print(f"## {category.value}")
            print()
            
            # 按重要性排序
            skills.sort(key=lambda s: s.importance, reverse=True)
            
            for skill in skills:
                priority = "高优先级" if skill.importance >= 4 else "中优先级"
                print(f"  • {skill.name} ({priority})")
            
            print()
        
        # 时间估算
        total_skills = len(skills_to_learn)
        months = total_skills * 0.5  # 假设每个技能0.5个月
        print(f"预计学习时间：{months:.0f}个月")
        print()

# 使用示例
planner = CareerPlanner()

# 打印技能树
planner.print_skill_tree()

# 生成学习计划
print("\n" + "="*80 + "\n")
planner.generate_learning_plan(Level.MIDDLE, Level.SENIOR)
```

```markdown
# 费曼学习法：最高效的学习方式

## 什么是费曼学习法？

以诺贝尔物理学奖得主Richard Feynman命名，
核心理念：**用简单的语言解释复杂的概念**

## 4个步骤

### 步骤1：选择概念
选择你要学习的概念，如"RAG系统"

### 步骤2：教给别人
假装你在教一个小白，用简单的语言解释：

"RAG系统就像是一个带着记事本的AI助手。

当你问它问题时：
1. 它先在记事本里翻找相关内容（检索）
2. 然后结合问题和找到的内容（增强）
3. 最后给你一个准确的回答（生成）

就像考试时可以翻书，答案更准确！"

### 步骤3：发现知识盲区
如果你无法用简单语言解释，说明你还不够理解。

比如：
• 向量检索的原理是什么？
• 为什么要做Embedding？
• 相似度怎么计算？

回去重新学习这些部分。

### 步骤4：简化和类比
用类比让概念更容易理解：

"Embedding就像把文章压缩成一个指纹，
 相似的文章指纹相似，
 向量检索就是找指纹最像的文章。"
```
